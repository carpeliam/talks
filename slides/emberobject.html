Ember.Object


  App.Person = Ember.Object.extend({
    // these will be supplied by `create`
    firstName: null,
    lastName: null,

    fullName: function() {
      return this.get('firstName') + ' ' + this.get('lastName');
    }.property('firstName', 'lastName')
  });

  var ironMan = App.Person.create({
    firstName: "Tony",
    lastName:  "Stark"
  });

  ironMan.get('fullName'); // "Tony Stark"

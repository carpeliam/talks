Ember.Object (CoffeeScript)


  App.Person = Ember.Object.extend
    # these will be supplied by `create`
    firstName: null
    lastName: null
    fullName: ( ->
      "#{@get('firstName')} #{@get('lastName')}"
    ).property('firstName', 'lastName')

  ironMan = App.Person.create
    firstName: 'Tony'
    lastName: 'Stark'

  ironMan.get 'fullName' # "Tony Stark"
